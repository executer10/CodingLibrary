세부 내용 정의

특징

  1. 파이썬 내장 함수는 다양한 데이터 타입을 지원합니다.
  2. 함수 호출이 간단하고 직관적입니다.

사용 이유

  1. 코드의 가독성을 높이고 유지보수를 용이하게 합니다.
  2. 복잡한 연산을 간단하게 처리할 수 있습니다.

print('Hello')  # Hello 출력

# name = input('이름을 입력하세요: ')

print(len([1, 2, 3]))  # 3

print(float('3.14'))  # 3.14

print(t)  # (1, 2, 3)

print(list(m))  # ['1', '2', '3']

result = filter(lambda x: x > 0, [-1, 0, 1])

print(issubclass(bool, int))  # True

# getattr() 예제
class Example:
    attr = 10
print(getattr(Example, 'attr'))  # 10

# setattr() 예제
class Example:
    pass
obj = Example()
setattr(obj, 'attr', 10)
print(obj.attr)  # 10

# hasattr() 예제
class Example:
    attr = 10
print(hasattr(Example, 'attr'))  # True

# delattr() 예제
class Example:
    attr = 10
delattr(Example, 'attr')
print(hasattr(Example, 'attr'))  # False

# property() 예제
class C:
    def __init__(self):
        self._x = None
    def getx(self):
        return self._x
    def setx(self, value):
        self._x = value
    def delx(self):
        del self._x
    x = property(getx, setx, delx, "I'm the 'x' property.")
c = C()
c.x = 5
print(c.x)  # 5

# staticmethod() 예제
class MyClass:
    @staticmethod
    def static_method():
        return "static method"
print(MyClass.static_method())  # "static method"

# classmethod() 예제
class MyClass:
    @classmethod
    def class_method(cls):
        return "class method"
print(MyClass.class_method())  # "class method"

# super() 예제
class Base:
    def method(self):
        return "Base method"
class Derived(Base):
    def method(self):
        return super().method() + " and Derived method"
d = Derived()
print(d.method())  # "Base method and Derived method"

# object() 예제
obj = object()
print(obj)  # 새로운 객체

# bytes() 예제
b = bytes(5)
print(b)  # 바이트 객체

# bytearray() 예제
ba = bytearray(5)
print(ba)  # 바이트 배열 객체

# memoryview() 예제
mv = memoryview(b'abc')
print(mv[0])  # 97

# frozenset() 예제
fs = frozenset([1, 2, 3])
print(fs)  # frozenset({1, 2, 3})

# complex() 예제
c = complex(1, 2)
print(c)  # (1+2j)

# ascii() 예제
print(ascii('파이썬'))  # "'\\ud30c\\uc774\\uc36c'"

# repr() 예제
print(repr(10))  # '10'

# globals() 예제
print(globals())  # 전역 변수 사전

# locals() 예제
def func():
    a = 1
    b = 2
    print(locals())
func()  # {'a': 1, 'b': 2}

# vars() 예제
class Example:
    def __init__(self):
        self.attr = 10
e = Example()
print(vars(e))  # {'attr': 10}

# compile() 예제
code = compile('print(1)', '', 'exec')
exec(code)  # 1 출력

# __import__() 예제
os = __import__('os')
print(os.name)  # 'posix' 또는 'nt'

# next() 예제
iterator = iter([1, 2, 3])
print(next(iterator))  # 1

# iter() 예제
iterator = iter([1, 2, 3])
print(next(iterator))  # 1

# reversed() 예제
r = reversed([1, 2, 3])
print(list(r))  # [3, 2, 1]

# slice() 예제
s = slice(1, 5, 2)
print(s)  # slice(1, 5, 2)

# format() 예제
print(format(3.14159, '.2f'))  # '3.14'

# hash() 예제
print(hash('abc'))  # 해시 값

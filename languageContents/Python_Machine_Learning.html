세부 내용

  1. 머신러닝이란?
    • 개념 정의

      머신러닝은 컴퓨터가 명시적으로 프로그래밍되지 않고도 데이터로부터 학습하고 자동으로 개선되는 인공지능의 한 분야입니다.

    • 작동 원리

      마치 아이가 경험을 통해 배우듯이, 머신러닝 알고리즘은 대량의 데이터를 통해 패턴을 학습하고 미래의 결과를 예측합니다.

    • 실제 활용 사례
      • 넷플릭스 영화 추천 시스템
      • 스마트폰 얼굴 인식
      • 자율주행 자동차의 장애물 감지
  2. 머신러닝 학습 방법
    • 지도 학습 (Supervised Learning)
      • 정의: 이미 레이블링된 데이터로 학습
      • 예시: 스팸 메일 분류, 주택 가격 예측
      • 주요 알고리즘: 선형 회귀, 로지스틱 회귀, 결정 트리
    • 비지도 학습 (Unsupervised Learning)
      • 정의: 레이블이 없는 데이터에서 패턴 발견
      • 예시: 고객 세분화, 추천 시스템
      • 주요 알고리즘: K-means 클러스터링, PCA
    • 강화 학습 (Reinforcement Learning)
      • 정의: 환경과 상호작용하며 최적의 행동 학습
      • 예시: 알파고, 로봇 제어 시스템
      • 핵심 개념: 보상과 페널티를 통한 학습

    # 필요한 라이브러리 불러오기
import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import StandardScaler
from sklearn.neighbors import KNeighborsClassifier
from sklearn.metrics import accuracy_score

# 데이터 불러오기 (예: 아이리스 데이터셋)
from sklearn.datasets import load_iris
iris = load_iris()
데이터 = pd.DataFrame(data=iris.data, columns=iris.feature_names)
레이블 = pd.Series(iris.target)

# 학습용과 테스트용 데이터 나누기
학습용_데이터, 테스트용_데이터, 학습용_레이블, 테스트용_레이블 = train_test_split(데이터, 레이블, test_size=0.2, random_state=42)

# 데이터 표준화
스케일러 = StandardScaler()
학습용_데이터 = 스케일러.fit_transform(학습용_데이터)
테스트용_데이터 = 스케일러.transform(테스트용_데이터)

# K-최근접 이웃 분류기 생성 및 학습
모델 = KNeighborsClassifier(n_neighbors=3)
모델.fit(학습용_데이터, 학습용_레이블)

# 예측 및 정확도 평가
예측_레이블 = 모델.predict(테스트용_데이터)
정확도 = accuracy_score(테스트용_레이블, 예측_레이블)
print(f'모델 정확도: {정확도 * 100:.2f}%')
  • 고급: 강화 학습을 통한 게임 에이전트 개발 
    # 필요한 라이브러리 불러오기
import gym
import numpy as np
import tensorflow as tf
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import Dense
from tensorflow.keras.optimizers import Adam

# 환경 설정 (예: CartPole-v1)
환경 = gym.make('CartPole-v1')
상태_크기 = 환경.observation_space.shape[0]
행동_크기 = 환경.action_space.n

# Q-네트워크 모델 생성
모델 = Sequential()
모델.add(Dense(24, input_dim=상태_크기, activation='relu'))
모델.add(Dense(24, activation='relu'))
모델.add(Dense(행동_크기, activation='linear'))
모델.compile(loss='mse', optimizer=Adam(lr=0.001))

# 하이퍼파라미터 설정
에피소드 = 1000
감가율 = 0.95
학습률 = 0.001
탐험률 = 1.0
탐험률_감소 = 0.995
최소_탐험률 = 0.01
배치_크기 = 32
메모리 = []

# 강화 학습 수행
for 에피소드_번호 in range(에피소드):
    상태 = 환경.reset()
    상태 = np.reshape(상태, [1, 상태_크기])
    총_보상 = 0
    
    while True:
        # 탐험 또는 활용 결정
        if np.random.rand() <= 탐험률:
            행동 = np.random.choice(행동_크기)
        else:
            행동_값 = 모델.predict(상태)
            행동 = np.argmax(행동_값[0])
        
        # 행동 수행 및 다음 상태, 보상, 종료 여부 관찰
        다음_상태, 보상, 종료, _ = 환경.step(행동)
        다음_상태 = np.reshape(다음_상태, [1, 상태_크기])
        총_보상 += 보상
        
        # 메모리에 저장
        메모리.append((상태, 행동, 보상, 다음_상태, 종료))
        상태 = 다음_상태
        
        # 종료 시 에피소드 종료
        if 종료:
            print(f'에피소드: {에피소드_번호}, 총 보상: {총_보상}, 탐험률: {탐험률:.2f}')
            break
        
        # 경험 재생 및 모델 학습
        if len(메모리) > 배치_크기:
            배치 = np.random.choice(len(메모리), 배치_크기)
            for 상태, 행동, 보상, 다음_상태, 종료 in [메모리[i] for i in 배치]:
                목표 = 보상
                if not 종료:
                    목표 += 감가율 * np.amax(모델.predict(다음_상태)[0])
                목표_값 = 모델.predict(상태)
                목표_값[0][행동] = 목표
                모델.fit(상태, 목표_값, epochs=1, verbose=0)
        
        # 탐험률 감소
        if 탐험률 > 최소_탐험률:
            탐험률 *= 탐험률_감소
